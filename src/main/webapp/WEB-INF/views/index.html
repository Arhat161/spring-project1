Создать таблицу Person (читатели)
для PostgresSQL 14

CREATE TABLE Person ( person_id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, person_fio varchar (100) NOT NULL UNIQUE, person_year_born int check ( person_year_born > 1900 ) NOT NULL );

Создать таблицу Book (книги)
для PostgresSQL 14

CREATE TABLE Book ( book_id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, person_id int REFERENCES Person(person_id) ON DELETE SET NULL, book_name varchar (100) NOT NULL, book_author varchar (100) NOT NULL, book_year int check ( book_year > 0 ) NOT NULL );
